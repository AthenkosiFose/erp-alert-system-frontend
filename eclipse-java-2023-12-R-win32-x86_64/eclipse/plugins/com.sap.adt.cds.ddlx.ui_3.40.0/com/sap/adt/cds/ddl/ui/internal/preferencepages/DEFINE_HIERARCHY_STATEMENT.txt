define hierarchy sample_define_hierarchy
  with parameters
    from_param : abap.dats,
    to_param : abap.dats,
    depth_param : abap.int4,
    directory_param : abap.int4
  as parent child hierarchy(
    source sflight
    child to parent association flights
    period from from_param to until_param
    valid from  $parameters.from_param to $parameters.to_param
    directory carriers filter by carrid = $parameters.directory_param
    start where
      carrid = 'AA'
    siblings order by
      carrid   ascending,
      carrname descending
    depth $parameters.depth_param
    nodetype name
    multiple parents allowed
    orphans ignore
    cycles breakup
  )
{
  carrid,
  connid
}
